 (defun backup (to &key (snap *snapshot*))
   (let ((installed (let (result)
 		     (dolist (i (directory ni::*src-directory* :wild "*.files" :absolute t))
 		       (setq result (append result (ni::data-read-from-file i))))
 		     (mapcar (lambda (x) (map-backslash-to-slash x)) result)))
 	(base (mapcar (lambda (x) (concat (si:system-root) x))
 		      (remove-if (lambda (x)
 				   (or
 				    (eql (char x 0) #\.)
 				    (string-match "XTAGS$" x)
 				    (string-match "/#[^#/]+#$" x)))
 				 (directory (si:system-root) :recursive t :file-only t))))
 	(builtin snap))
     (dolist (i (dolist (i installed base)
 		 (delete i base :test 'equal)
 		 (delete (concat (si:system-root) i) base :test 'equal))))
     (setq builtin (mapcar (lambda (x) (cons (concat (si:system-root) (car x)) (cadr x))) builtin))
     (dolist (i builtin)
       (when (or (not (file-exist-p (car i)))
 		(equal (ni::md5sum (car i)) (cdr i)))
 	(delete (car i) base :test 'equal)))
     (delete (si:dump-image-path) base :test 'equal)
     (let ((destdir to)
 	  dest)
       (dolist (i base)
 	(setq dest (merge-pathnames (substring i (length (si:system-root)) nil) destdir))
 	(create-directory (directory-namestring dest) :if-exists :skip)
 	(copy-file i dest)))
     base))
 
 (defun make-snapshot (&optional basedir)
   (let ((dir (or basedir (si:system-root))))
     (dolist (i (directory dir :recursive t :file-only t))
       (format t "(\"~A\" \"~A\")~%" i (ni::md5sum (concat dir i))))))
